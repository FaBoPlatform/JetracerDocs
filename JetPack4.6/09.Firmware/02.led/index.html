    #include "SPI.h"

void startBit() {
  byte start = 0x00;
  SPI.transfer(start);
  SPI.transfer(start);
  SPI.transfer(start);
  SPI.transfer(start);
}

void endBit() {
  byte end = 0x11;
  SPI.transfer(end);
  SPI.transfer(end);
  SPI.transfer(end);
  SPI.transfer(end);
}

void setRGB(short r, short g, short b) {
  SPI.transfer(0xEF);
  SPI.transfer(r);
  SPI.transfer(g);
  SPI.transfer(b);
}

void setup() {
  SPI.begin();
}

void loop() {
  for(int i = 0; i < 255; i++) {
    startBit();
    setRGB(255,   i,   0);
    setRGB(255,   i, 255);
    setRGB(255, 255,   i);
    setRGB(255,   0,   i); 
    setRGB(i,   255,   0);
    setRGB(i,   255,   255);
    endBit();
    delay(100);
  }
}
